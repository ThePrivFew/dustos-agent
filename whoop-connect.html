// Add to dustos-proxy.thewattleroad.workers.dev

// Environment variables to add:
// WHOOP_CLIENT_ID      — from app.whoop.com/developer
// WHOOP_CLIENT_SECRET  — from app.whoop.com/developer
// WHOOP_REDIRECT_URI   — https://theprivfew.github.io/dustos-agent/whoop-connect.html

// Route 1: Token exchange
if (url.pathname === '/whoop/token') {
  const body = await request.json();
  const params = new URLSearchParams({
    grant_type: 'authorization_code',
    code: body.code,
    redirect_uri: env.WHOOP_REDIRECT_URI,
    client_id: env.WHOOP_CLIENT_ID,
    client_secret: env.WHOOP_CLIENT_SECRET,
    code_verifier: body.code_verifier,
  });
  const resp = await fetch('https://api.prod.whoop.com/oauth/oauth2/token', {
    method: 'POST',
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
    body: params.toString(),
  });
  const data = await resp.json();
  return new Response(JSON.stringify(data), {
    headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': env.ALLOWED_ORIGIN }
  });
}

// Route 2: API proxy (all Whoop API calls)
if (url.pathname.startsWith('/whoop/api/')) {
  const whoopPath = url.pathname.replace('/whoop/api', '');
  const whoopUrl = 'https://api.prod.whoop.com/developer/v1' + whoopPath + url.search;
  const token = request.headers.get('X-Whoop-Token');
  const resp = await fetch(whoopUrl, {
    headers: { 'Authorization': 'Bearer ' + token }
  });
  const data = await resp.json();
  return new Response(JSON.stringify(data), {
    headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': env.ALLOWED_ORIGIN }
  });
}
