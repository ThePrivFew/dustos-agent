// Environment variables to add:
// SPOTIFY_CLIENT_ID      — from developer.spotify.com
// SPOTIFY_CLIENT_SECRET  — from developer.spotify.com
// SPOTIFY_REDIRECT_URI   — https://theprivfew.github.io/dustos-agent/spotify-connect.html

// Route 1: Token exchange (PKCE — no client secret in browser)
if (url.pathname === '/spotify/token') {
  const body = await request.json();
  const params = new URLSearchParams({
    grant_type: 'authorization_code',
    code: body.code,
    redirect_uri: env.SPOTIFY_REDIRECT_URI,
    client_id: env.SPOTIFY_CLIENT_ID,
    code_verifier: body.code_verifier,
  });
  const resp = await fetch('https://accounts.spotify.com/api/token', {
    method: 'POST',
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
    body: params.toString(),
  });
  const data = await resp.json();
  return new Response(JSON.stringify(data), {
    headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': env.ALLOWED_ORIGIN }
  });
}

// Route 2: API proxy
if (url.pathname.startsWith('/spotify/api/')) {
  const spotifyPath = url.pathname.replace('/spotify/api', '');
  const spotifyUrl = 'https://api.spotify.com/v1' + spotifyPath + url.search;
  const token = request.headers.get('X-Spotify-Token');
  const resp = await fetch(spotifyUrl, {
    headers: { 'Authorization': 'Bearer ' + token }
  });
  const data = await resp.json();
  return new Response(JSON.stringify(data), {
    headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': env.ALLOWED_ORIGIN }
  });
}
